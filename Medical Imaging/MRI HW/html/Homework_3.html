Plotting

Time span for plotting.

t = 0:0.000001:.200;
% Define the equations.
Mx = exp(-t./T2).*cos(gamma*B.*t);
My = -exp(-t./T2).*sin(gamma*B.*t);
Mz = (1-exp(-t./T1));
% Plot Mx and My
figure(1); hold on;
subplot(2,1,1); plot(t, Mx, 'r')
title('M_x(t)')
xlabel('t')
ylabel('Magnetic Field Magnitude')
subplot(2,1,2); plot(t, My, 'g')
title('M_y(t)')
xlabel('t')
ylabel('Magnetic Field Magnitude')
