% Problem 1.15(a)

Fs = 5e3;
F0 = [0.5 2 3 4.5].*1e3;

n = 0:1:99;
t = n./Fs;

figure(1);
for i = 1:length(F0)
    x = sin(2*pi*F0(i)*t);

    subplot(2,2, i);
    stem(n, x);
    xlabel('n');
    title([' F_0 = ', num2str(F0(i)), 'Hz']);

end
