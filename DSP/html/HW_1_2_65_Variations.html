% Generate a Period 127 Signal as described in the text.
seq = [1 0 0 0 0 0 0];
output = [];
for i = 1:2^7-1

   feedback = xor(seq(1), seq(end));
   output = [output seq(end)];
   seq = [feedback seq(1:end-1)];

end
output =  2*output-1;

HW1_2_65(0.01, output, 9, 'Period 127 Sequence');
